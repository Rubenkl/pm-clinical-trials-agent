Portfolio Manager Implementation


from openai.agents import Agent, Context, HandoffResult

class PortfolioManager(Agent):
    def __init__(self):
        super().__init__(name="portfolio_manager")
        self.query_analyzer = QueryAnalyzerAgent()
        self.data_verifier = DataVerifierAgent()
        self.deviation_detector = DeviationAgent()
    
    async def run(self, context: Context) -> HandoffResult:
        """Orchestrate multi-agent workflow"""
        
        # Analyze incoming query
        analysis_result = await context.handoff_to_agent(
            self.query_analyzer,
            message="Analyze clinical data query"
        )
        
        # Determine next agent based on analysis
        if analysis_result.requires_sdv:
            return await context.handoff_to_agent(
                self.data_verifier,
                message=f"Verify data: {analysis_result.data}"
            )
        elif analysis_result.potential_deviation:
            return await context.handoff_to_agent(
                self.deviation_detector,
                message=f"Check deviation: {analysis_result.protocol}"
            )
        
        return HandoffResult(
            result=analysis_result,
            next_agent=None  # Workflow complete
        )
